module top_module (
    input clk,
    input aresetn,    // Asynchronous active-low reset
    input x,
    output z ); 
    reg [1:0] state, next;
    localparam A=0, B=1, C=2;
    always @(*) begin
        case (state)
            A: next = x ? B : A;
            B: next = x ? B : C;
            C: next = x ? B : A;
        endcase
    end
    assign z = (state == C && x);//depends on state and input
    always @(posedge clk or negedge aresetn) begin
        if (!aresetn)
            state <= A;
        else
            state <= next;
    end
endmodule
