module top_module (
    input clk,
    input d,
    output q
);
    reg pos_q,neg_q;
    //we need to capture q on both edges
    always@(posedge clk)begin
        pos_q<=d;
    end
    always@(negedge clk)begin
        neg_q<=d;
    end
    assign q=clk?pos_q:neg_q;//2:1 mux logic
endmodule
