module top_module (
    input [3:1] y,
    input w,
    output reg Y2);
    
    always@(*)begin//we draw table and we find the middle bit of next stages
        case(y)
            3'b000:Y2=0;//for A n D Y2=0
            3'b001:Y2=1;//for B n F it is 1
            3'b010:Y2=w;//c and e->w
            3'b011:Y2=0;
            3'b100:Y2=w;
            3'b101:Y2=1;
            default:Y2=0;
        endcase
    end
endmodule
